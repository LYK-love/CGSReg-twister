# Offline TWISTER WM training on the Atari100K repro online ReplayBuffer.
#
# Purpose: isolate the dataset effect by replacing the DIAMOND-converted static
# replay with TWISTER's own Atari100K Pong replay buffer while keeping AC-CPC on.
#
# Dataset:
#   callbacks/atari100k/atari100k-pong/ReplayBuffer
# Source run:
#   callbacks/atari100k/atari100k-pong/checkpoints_epoch_50_step_100000.ckpt
#
# Protocol:
#   - base TWISTER WM
#   - AC-CPC enabled; no loss_contrastive_scale override
#   - mask1 spatial regularizer
#   - spatial weights: 0.0, 0.01, 0.1, 1.0
#   - 100k offline optimizer steps
#   - batch size 16, batch length 64
#   - native ReplayBuffer.load/sample path, matching dyna-style TWISTER training
#   - mask1 derived from consecutive-frame motion for spatial regularization
#
# Launch from the TWISTER repository root:
#
#   conda activate twister
#   cd "$HOME/projects/twister"
#   tiny-exp-scheduler run scripts/experiments/pong_offline_replay_ac_cpc_w_sweep.commands.txt \
#     --cuda-devices auto \
#     --logs-dir runs/pong_offline_replay_ac_cpc_w_sweep/scheduler_logs \
#     --verbose \
#     --keep-job-tabs
