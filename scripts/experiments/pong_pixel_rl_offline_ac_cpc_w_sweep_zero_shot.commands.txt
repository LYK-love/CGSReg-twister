# Zero-shot RL in TWISTER offline-trained Pong WMs with AC-CPC enabled.
#
# Spatial weights: [0.0, 0.003, 0.005, 0.01, 0.02, 0.05, 0.1, 1.0]
# WM checkpoints are restored from the Box package:
#   box:offline_training/twister/twister_offline_ac_cpc_spatial_wm_ckpts_w0_to_w10_20260612T144102/
# Local checkpoint root after extraction:
#   external_wm_ckpts/offline_ac_cpc_spatial/
#
# Protocol: common rl-in-pixel-env Pong WM protocol.
#   - 20k actor-critic updates
#   - 64 envs
#   - backup length 15
#   - horizon 512
#   - reset=real
#   - reward quantization threshold 0.1
#   - real Atari eval every 2000 AC updates, 5 episodes
#   - scalar log every 1000 AC updates
#   - policy checkpoint/video every 10000 AC updates
#   - CPU math thread pools limited to 1 thread per job to reduce host contention
#
# Launch on pandaria from the TWISTER repository root. Use explicit GPU ids;
# tiny-exp-scheduler runs one job per listed GPU. If CPU contention is still
# high, reduce the list to 0,1 instead of changing the experiment protocol.
#
#   conda activate twister
#   cd "$HOME/projects/twister"
#   tiny-exp-scheduler run scripts/experiments/pong_pixel_rl_offline_ac_cpc_w_sweep_zero_shot.commands.txt \
#     --cuda-devices 0,1,2,3 \
#     --logs-dir pong_pixel_rl_in_env/scheduler_logs/offline_ac_cpc_w_sweep_zero_shot_pandaria \
#     --verbose \
#     --keep-job-tabs
